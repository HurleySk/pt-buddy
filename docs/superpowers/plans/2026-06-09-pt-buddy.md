# PT Buddy Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a SuuntoPlus Sports App for the Suunto Race S that tracks PT exercises with a rep counter, hold timer, and rest timer — including bilateral (L/R) support.

**Architecture:** Pure state machine in `src/state.js` with all logic as tested, side-effect-free functions. Each function takes state and returns `{ state, effects }`. A build script wraps this into a SuuntoPlus-compatible `main.js` that bridges button events and display updates. Display is a single `display.html` with CSS color themes for each mode.

**Tech Stack:** JavaScript (ES modules for dev/test, ES5-compatible output for watch), Vitest (testing), SuuntoPlus Editor (build/deploy)

---

## File Structure

```
pt-buddy/
├── src/
│   └── state.js              # Pure state machine (all logic, no side effects)
├── tests/
│   └── state.test.js         # Unit tests for state machine
├── scripts/
│   └── build.js              # Concatenates src/ → main.js for SuuntoPlus
├── main.js                   # Built output (SuuntoPlus entry point)
├── display.html              # Watch face template (3 color-coded screens)
├── manifest.json             # SuuntoPlus app manifest
├── package.json              # Dev dependencies (vitest)
└── docs/superpowers/specs/   # Design spec (already exists)
```

**Key design decision:** The state machine is pure — every function takes a state object and returns `{ state, effects }` where effects are data (e.g., `{ type: "vibrate", pattern: "short", count: 3 }`). The SuuntoPlus bridge in `main.js` applies these effects to the real watch APIs. This keeps all logic testable without mocking watch hardware.

function transitionToRest(state, effects) {
  var next = Object.assign({}, state);
  next.previousMode = state.mode;
  next.mode = "rest";
  next.timerRemaining = state.restDuration;
  next.timerRunning = true;
  next.reps = 0;
  return { state: next, effects: effects };
}

export function transitionTap(state) {
  var next = Object.assign({}, state);
  var effects = [];

  if (state.mode === "count") {
    if (state.reps === 0) {
      return { state: state, effects: effects };
    }
    return transitionToRest(next, effects);
  }

  return { state: state, effects: effects };
}
```

function completeRest(state, effects) {
  var next = Object.assign({}, state);
  next.mode = state.previousMode;
  next.set = state.set + 1;
  next.reps = 0;
  next.timerRunning = false;
  next.timerRemaining = 0;
  next.activeSide = "L";
  return { state: next, effects: effects };
}

export function tick(state) {
  if (!state.timerRunning) {
    return { state: state, effects: [] };
  }

  var next = Object.assign({}, state);
  var effects = [];

  next.timerRemaining = state.timerRemaining - 1;

  if (next.timerRemaining <= 0) {
    next.timerRunning = false;

    if (state.mode === "rest") {
      return completeRest(next, effects);
    }
  }

  return { state: next, effects: effects };
}
```

export function actionLongPress(state) {
  if (!isFresh(state)) {
    return { state: state, effects: [] };
  }

  var next = Object.assign({}, state);

  if (state.mode === "count") {
    next.mode = "hold";
  } else if (state.mode === "hold") {
    next.mode = "count";
  }

  next.reps = 0;

var appState;
var lastTickTime = 0;

function applyEffects(effects) {
  for (var i = 0; i < effects.length; i++) {
    var e = effects[i];
    if (e.type === "vibrate") {
      // SuuntoPlus vibration API — verify exact method name in editor docs
      if (typeof SUUNTO !== "undefined" && SUUNTO.alarms) {
        for (var j = 0; j < e.count; j++) {
          SUUNTO.alarms.vibrate(e.pattern);
        }
      }
    }
  }

function processResult(result) {
  appState = result.state;
  applyEffects(result.effects);
}

function updateDisplay() {
  if (typeof SUUNTO === "undefined") return;

  var modeLabels = { count: "COUNT", hold: "HOLD", rest: "REST" };
  var s = appState;

  // Output variables — names must match manifest.json outputs and display.html template
  SUUNTO.outputs.mode_label = modeLabels[s.mode] || "";

  if (s.mode === "count") {
    SUUNTO.outputs.main_value = String(s.reps);
    SUUNTO.outputs.sub_label = "reps";
  } else {
    var mins = Math.floor(s.timerRemaining / 60);
    var secs = s.timerRemaining % 60;
    SUUNTO.outputs.main_value = mins + ":" + (secs < 10 ? "0" : "") + secs;
    SUUNTO.outputs.sub_label = "remaining";
  }

  if (s.mode === "rest") {
    SUUNTO.outputs.set_label = "Next: Set " + (s.set + 1);
  } else {
    SUUNTO.outputs.set_label = "Set " + s.set;
  }

  var themeColors = { count: "#4CAF50", hold: "#FF9800", rest: "#2196F3" };
  SUUNTO.outputs.theme_color = themeColors[s.mode] || "#fff";
  SUUNTO.outputs.bilateral = s.bilateral;
  SUUNTO.outputs.active_side = s.activeSide;

  if (s.mode === "hold" || s.mode === "rest") {
    var total = s.mode === "hold" ? s.holdDuration : s.restDuration;
    SUUNTO.outputs.progress = total > 0 ? s.timerRemaining / total : 0;
  } else {
    SUUNTO.outputs.progress = 0;
  }
}

function onLoad() {
  appState = createState();
  updateDisplay();
}

function evaluate(context) {
  // Timer tick — check if a second has elapsed
  // The exact time source depends on the SuuntoPlus API — verify in editor docs.
  // context.time, SUUNTO.time, or SUUNTO.duration are common candidates.
  var now = 0;
  if (context && context.time != null) {
    now = context.time;
  } else if (typeof SUUNTO !== "undefined" && SUUNTO.time != null) {
    now = SUUNTO.time;
  }

  if (now - lastTickTime >= 1) {
    lastTickTime = now;
    if (appState.timerRunning) {
      processResult(tick(appState));
    }
  }

  // Button handling — verify event property names in editor docs
  if (context && context.button) {
    if (context.button.action === "tap") {
      processResult(actionTap(appState));
    } else if (context.button.action === "longpress") {
      processResult(actionLongPress(appState));
    } else if (context.button.transition === "tap") {
      processResult(transitionTap(appState));
    } else if (context.button.transition === "longpress") {
      processResult(transitionLongPress(appState));
    }
  }

  updateDisplay();
}
